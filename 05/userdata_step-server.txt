#!/bin/bash
cp /usr/share/zoneinfo/Asia/Tokyo /etc/localtime
yum install -q -y http://rdo.fedorapeople.org/rdo-release.rpm
yum install -q -y python-novaclient \
 python-neutronclient \
 python-glanceclient \
 python-cinderclient \
 python-swiftclient \
 python-keystoneclient
cat << EOF > /root/openrc
export OS_AUTH_URL=http://192.168.100.10:5000/v2/
export OS_REGION_NAME=RegionOne
export OS_TENANT_NAME=SNSApp
export OS_USERNAME=snsapp-infra-user
export OS_PASSWORD=passw0rd
EOF
